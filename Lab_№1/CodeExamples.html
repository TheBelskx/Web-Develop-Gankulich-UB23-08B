Примеры кода на C++

  • Пример 1: Hello World

    #include <iostream>

int main() {
    std::cout << "Hello, World!" << std::endl;
    return 0;
}

  • Пример 2: Сложение двух чисел

    #include <iostream>

int main() {
    int a, b;
    std::cout << "Введите два числа: ";
    std::cin >> a >> b;
    std::cout << "Сумма: " << (a + b) << std::endl;
    return 0;
}

  • Пример 3: Функция факториала

    #include <iostream>

int factorial(int n) {
    return (n == 1 || n == 0) ? 1 : factorial(n - 1) * n;
}

int main() {
    int num;
    std::cout << "Введите число: ";
    std::cin >> num;
    std::cout << "Факториал: " << factorial(num) << std::endl;
    return 0;
}
